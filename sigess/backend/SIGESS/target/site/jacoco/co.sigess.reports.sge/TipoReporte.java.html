/**
 *
 * @author fmoreno
 */
public enum TipoReporte {
    DOCX("docx"),
    XLSX("xlsx"),
    PDF("pdf");
    
    private final String extension;

    private TipoReporte(String extension) {
        this.extension = extension;
    }

    public String getExtension() {
        return extension;
    }
    
    
}
