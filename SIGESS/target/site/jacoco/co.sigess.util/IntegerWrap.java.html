package co.sigess.util;

/**
 *
 * @author fmoreno
 */
public class IntegerWrap {

    private int value;

    public void increment(int increment) {
        value += increment;
    }

    public int getValue() {
        return value;
    }

    public void setValue(int value) {
        this.value = value;
    }

}
