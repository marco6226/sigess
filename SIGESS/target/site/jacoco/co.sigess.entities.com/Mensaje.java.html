    private String mensaje;

    public Mensaje() {

    public Mensaje(String mensaje) {
        this.mensaje = mensaje;

        this.mensaje = mensaje;
        this.detalle = detalle;
    }

    public Mensaje(String mensaje, String detalle, TipoMensaje tipoMensaje) {
        this.mensaje = mensaje;
